ImageBase64 图片处理方法，将用户上传的图片等比例缩放至设定值（取图片中间位置);
script脚本：
    var baseImageArea=document.getElementById("baseImageArea");
    var imageBase64=ImageBase64(200,200);
    imageBase64.setCallback(function(_type,_message){
    switch(_type){
    case "success":
    baseImageArea.setAttribute("src",_message);
    }
    })
图片预览：
特殊说明：
此对象提供丰富的回调函数，如无要求可不进行处理，其中error和success事件是必要的处理事件。
代码说明：
var imageBase64 = ImageBase64(200,200);
//实例化
imageBase64.setCallback(function(_type,_message){
switch(_type){
case "progress":
//读取文件过程中 注：_message为进度百分比，arguments[2]为已加载数据大小，arguments[3]为总大小
break;
case "abort":
//已取消文件的读取操作 注：_message为进度百分比，arguments[2]为已加载数据大小，arguments[3]为总大小
break;
case "complete":
//文件读取完毕
break;
case "start":
//开始读取文件
break;
case "abort":
//已取消文件的读取操作
break;
case "error":
//文件读取失败
break;
case "success":
//文件读取成功 注：message为base64URL,可直接放入图片的src
break;
})
imageBase64.open();
//打开选图窗口
.setFormat("image/png");
//设置编码格式，image/png、image/jpeg、image/webp
.getFormat();
//获取编码格式，image/png、image/jpeg、image/webp
.setEncoderOptions(0.92);
//获取压缩质量，当格式为image/jpeg、image/webp时有效，取值范围0~1
.getEncoderOptions();
//获取压缩质量

AjaxPlus封装ajax方法，把数据交换协议进行处理，使用时无需处理协议部分。
使用说明：
  • 此方法依赖jQuery
  • 常搭配loadingView使用
代码说明：
var LoadingView=null;
声明LoadingView变量
AjaxPlus({
url:'',
//请求数据的地址
data:{data:"csacsa"},
//传递的参数
errorMessage:"",
//发生错误时的提示信息，可以忽略，忽略则系统自动处理
beforeSend:function(){ LoadingView=new LoadingBox(); },
//创建并显示loadingbox
errorBefore:function(){},
//错误信息拦截，返回false系统不再进行处理
errorAfter:function(){},
//系统预处理错误提示结束后调用
success:function(_data){},
//数据页面加载成功
error:function(_xhr,_textStatus,_errorThrown){},
//一旦写了这个处理函数，errorBefore和errorAfter 都不会出发，需要自己处理错误信息
complete:function(_xhr,_success){ LoadingView.remove(); },
//不管ajax执行成功与否均调用此方法，并传递是否发生错误参数,并移除loadingView
loginError:function(){}，
//未登录的时候触发，返回false系统不会再继续处理错误信息
authorityError:function(){}
//没有权限的时候触发，返回false系统不会再继续处理错误信息
})

  • 简写方式 var loadingView=LoadingView(2000); //两秒消失
  • 插入到某个元素中，默认为body
  • 延时销毁（隐藏）
  • 改变背景色

RequestUrl用于获取URL中的参数。
使用说明：
var requestUrl=RequestUrl([url]);//创建对象（无new构造）,[url]为可选参数，默认为当前页面URL
例：
var requestUrl=RequestUrl("http://baidu.com?id=98&key=%E4%BD%A0%E5%A5%BD");//创建对象，参数可为空，默认为document.location
requestUrl.get("id"); //获取id的值，返回98
requestUrl.getString("key"); //获取key的值,同requestUrl.get，返回%E4%BD%A0%E5%A5%BD
requestUrl.getInt("id",0); //获取id的值，返回整数类型的返回值：98
requestUrl.getUnescape("key","123456"); //获取key的值并进行解码，返回：你好
注： 也可使用简写方式调用,如：requestUrl.id、requestUrl.key//此方法使用直接获取对象上的值，如果url上不存在此属性，则返回undefined
特殊说明：
也可使用简写方式调用,如：
requestUrl.id、requestUrl.key //此方法使用直接获取对象上的值，如果url上不存在此属性，则返回undefined
代码说明：

checkMode设置为true则下拉框为复选模式，回调数据将以逗号分隔。
代码说明：
